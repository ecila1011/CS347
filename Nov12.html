I created seperate JS files for each of the burger ingredients, and each had its own css file to create the image for that ingredient. Then, I had a Create Burger file that took in an object from the App.js to create the burger itself, and then displayed it. I have included both the important parts of the App.js file and the CreateBurger.js file, since they contain the bulk of the code.

App.js

        function App() {

            return (
              <div className="App">
                <h1>Choose Your Burger Ingredients</h1>
                <CreateBurger
                  burgerStack={[
                    'Lettuce',
                    'Tomatoes',
                    'Cheese',
                    'Lettuce',
                    'Burger',
                    'Tomatoes',
                    'Tomatoes'
                  ]}
                />
              </div>
            );
          }

CreateBurger.js

        export default function CreateBurger(props) {
            const BurgerStack = props.burgerStack.map((item, index) => {
              switch (item) {
                case 'Cheese':
                  return <Cheese key={index} />;
                case 'Lettuce':
                  return <Lettuce key={index} />;
                case 'Burger':
                  return <Burger key={index} />;
                case 'Tomatoes':
                  return <Tomatoes key={index} />;
                default:
                  return null;
              }
            });
          
            return (
              <div>
                <BreadUpper />
                {BurgerStack}
                <BreadLower />
              </div>
            );
          }
